48 GLuint CreateShader( string vertex, string fragment ){
49  /*
50  Tworzenie shadera wierzchołków i fragmentu
51  Zmienne w których przypisane będą ww shadery
52  */
53  GLuint VertexShaderID = glCreateShader( GL_VERTEX_SHADER );
54  GLuint FragmentShaderID = glCreateShader( GL_FRAGMENT_SHADER );
55 
56  /*
57  Zmienne dla pobierania tekstu z plików tekstowych z shaderami
58  */
59  string Line;
60  //vertex_shader:
61  string VertexShaderCode;
62  /*
63  Otwarcie pliku tekstowego do odczytu
64  */
65  ifstream VertexShaderStream( vertex.c_str() );
66  if( VertexShaderStream.good() ){
67  Line = "";
68  while( getline( VertexShaderStream, Line ) ){
69  VertexShaderCode += Line + "\n";
70  }
71  VertexShaderStream.close();
72  }
73  else{
74  cout<<"File problem: "<<vertex<<"\n";
75  return 0;
76  }
77 
78  //Load code fragment_shader:
79  string FragmentShaderCode;
80  ifstream FragmentShaderStream( fragment.c_str() );
81  if( FragmentShaderStream.good() ){
82  Line = "";
83  while( getline( FragmentShaderStream, Line ) ){
84  FragmentShaderCode += Line + "\n";
85  }
86  FragmentShaderStream.close();
87  }
88  else{
89  cout<<"File problem: "<<fragment<<"\n";
90  return 0;
91  }
92 
93  GLint Result;
94  GLint InfoLogLength;
95  string Error;
96 
97  const char *VertexShaderCodePointer = VertexShaderCode.c_str();
98  glShaderSource( VertexShaderID, 1, &VertexShaderCodePointer, 0 );
99  glCompileShader( VertexShaderID );
100  //Check vertex_shader:
101  Result = GL_FALSE;
102  glGetShaderiv( VertexShaderID, GL_COMPILE_STATUS, &Result );
103  if( Result == GL_FALSE ){
104  InfoLogLength = 0;
105  glGetShaderiv( VertexShaderID, GL_INFO_LOG_LENGTH, &InfoLogLength );
106  vector <GLchar> VertexShaderErrorMessage(InfoLogLength);
107  glGetShaderInfoLog( VertexShaderID, InfoLogLength, &InfoLogLength, &VertexShaderErrorMessage[0] );
108  Error = string( VertexShaderErrorMessage.begin(), VertexShaderErrorMessage.end() );
109  cout<<"Vertex shader: "<<Error<<"\n";
110  glDeleteShader( VertexShaderID );
111  return 0;
112  }
113 
114  //fragment_shader:
115  const char *FragmentShaderCodePointer = FragmentShaderCode.c_str();
116  glShaderSource( FragmentShaderID, 1, &FragmentShaderCodePointer, 0 );
117  glCompileShader( FragmentShaderID );
118  //Check fragment_shader:
119  Result = GL_FALSE;
120  glGetShaderiv( FragmentShaderID, GL_COMPILE_STATUS, &Result );
121  if( Result == GL_FALSE ){
122  InfoLogLength = 0;
123  glGetShaderiv( FragmentShaderID, GL_INFO_LOG_LENGTH, &InfoLogLength );
124  vector <GLchar> FragmentShaderErrorMessage(InfoLogLength);
125  glGetShaderInfoLog( FragmentShaderID, InfoLogLength, &InfoLogLength, &FragmentShaderErrorMessage[0] );
126  Error.clear();
127  Error = string( FragmentShaderErrorMessage.begin(), FragmentShaderErrorMessage.end() );
128  cout<<"Fragment shader: "<<Error<<"\n";
129  glDeleteShader( VertexShaderID );
130  glDeleteShader( FragmentShaderID );
131  return 0;
132  }
133 
134  Result = GL_FALSE;
135  InfoLogLength = 0;
136 
137  //Create program shader:
138  GLuint ProgramID = glCreateProgram();
139  glAttachShader( ProgramID, VertexShaderID );
140  glAttachShader( ProgramID, FragmentShaderID );
141  glLinkProgram( ProgramID );
142 
143  //Check program:
144  glGetProgramiv( ProgramID, GL_LINK_STATUS, &Result );
145  if( Result == GL_FALSE ){
146  glGetProgramiv( ProgramID, GL_INFO_LOG_LENGTH, &InfoLogLength );
147  vector <GLchar> ProgramErrorMessage(InfoLogLength);
148  glGetProgramInfoLog( ProgramID, InfoLogLength, &InfoLogLength, &ProgramErrorMessage[0] );
149  Error.clear();
150  Error = string( ProgramErrorMessage.begin(), ProgramErrorMessage.end() );
151  cout<<"program shader: "<<Error<<"\n";
152  glDeleteProgram( ProgramID );
153  glDeleteShader( VertexShaderID );
154  glDeleteShader( FragmentShaderID );
155  return 0;
156  }
157 
158  //delete shaders:
159  glDetachShader( ProgramID, VertexShaderID );
160  glDetachShader( ProgramID, FragmentShaderID );
161  glDeleteShader( VertexShaderID );
162  glDeleteShader( FragmentShaderID );
163 
164  return ProgramID;
165 }

168  ILenum error;
169  GLenum error_gl;
170  ILboolean success;
171  ILuint imgage_id;
172 
173  ilGenImages( 1, &imgage_id );
174  ilBindImage( imgage_id );
175  success = ilLoadImage( img_path_file );
176  if( ! success ){
177  error = ilGetError();
178  if( error != IL_NO_ERROR ){
179  cout<<"ilLoadImage ("<<img_path_file<<"): "<<iluErrorString( error )<<"\n";
180  image = 0;
181  ilDeleteImages( 1, &imgage_id );
182  return false;
183  }
184  cout<<"ilLoadImage ("<<img_path_file<<"): "<<ilGetError()<<"\n";
185  image = 0;
186  ilDeleteImages( 1, &imgage_id );
187  return false;
188  }
189 
190  GLint width = ilGetInteger( IL_IMAGE_WIDTH );
191  GLint height = ilGetInteger( IL_IMAGE_HEIGHT );
192  GLint type = ilGetInteger( IL_IMAGE_TYPE );
193  GLint format = ilGetInteger( IL_IMAGE_FORMAT );
194 
195  glGenTextures( 1, &image );
196  glBindTexture( GL_TEXTURE_2D, image );
197 
198  glTexParameteri( image, GL_TEXTURE_WRAP_S, GL_REPEAT );
199  glTexParameteri( image, GL_TEXTURE_WRAP_T, GL_REPEAT );
200 
201  glTexImage2D( GL_TEXTURE_2D, 0, format, width, height, 0, format, type, ilGetData() );
202  error_gl = glGetError();
203  if( error_gl != GL_NO_ERROR and error_gl != GL_INVALID_ENUM ){
204  cout<<"glTexImage2D ("<<img_path_file<<"): "<<gluErrorString( error_gl )<<"\n";
205  return false;
206  }
207 
208  glGenerateMipmap( GL_TEXTURE_2D );
209 
210  glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT );
211  glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT );
212  glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR );
213  glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR );
214 
215  glBindTexture(GL_TEXTURE_2D, 0);
216 
217  ilDeleteImages( 1, &imgage_id );
218  return true;
219 }

221 bool LoadAssimp( string file, vector <vec3> &vertices, vector <vec2> &uvs, vector <vec3> &normals, vector <GLuint> &indices ){
222  vertices.clear();
223  uvs.clear();
224  normals.clear();
225  indices.clear();
226  Assimp::Importer importer;
227  const aiScene* scene = importer.ReadFile( file.c_str(), 0 );
228  if( !scene ){
229  cout<<"importer.ReadFile ("<<file<<"): "<<importer.GetErrorString()<<"\n";
230  return false;
231  }
232  const aiMesh* mesh = scene->mMeshes[0];
233  unsigned int i = 0;
234  aiVector3D tmp;
235  //vertices:
236  vertices.reserve( mesh->mNumVertices );
237  for( i = 0; i < mesh->mNumVertices; ++i ){
238  tmp = mesh->mVertices[i];
239  vertices.push_back( vec3( tmp.x, tmp.y, tmp.z ) );
240  }
241  //uvs:
242  uvs.reserve( mesh->mNumVertices );
243  for( i = 0; i < mesh->mNumVertices; ++i ){
244  tmp = mesh->mTextureCoords[0][i];
245  //important!
246  //This is done because most images have the top y-axis inversed with OpenGL's top y-axis.
247  //or conver in shader
248  uvs.push_back( glm::vec2( tmp.x, 1.0 - tmp.y ) );
249  }
250  //normals:
251  normals.reserve( mesh->mNumVertices );
252  for( i = 0; i < mesh->mNumVertices; ++i ){
253  tmp = mesh->mNormals[i];
254  normals.push_back( vec3( tmp.x, tmp.y, tmp.z ) );
255  }
256  //indices:
257  indices.reserve( 3*mesh->mNumFaces );
258  for( i = 0; i < mesh->mNumFaces; ++i ){
259  indices.push_back( mesh->mFaces[i].mIndices[0] );
260  indices.push_back( mesh->mFaces[i].mIndices[1] );
261  indices.push_back( mesh->mFaces[i].mIndices[2] );
262  }
263  return true;
264 }
