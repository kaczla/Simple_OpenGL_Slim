469  /*
470  Gry wszystko jest załadowane, to zacznij odbierać sygnały zdarzeń w SDL2 (klawiatury i myszy).
471  */
472  while( this->CheckInit ){
473  /*
474  Wykonuj dopóki jest jakieś zdarzenie w SDL2.
475  */
476  while( SDL_PollEvent( & this->Event ) ){
477  /*
478  Podział na typt zdarzeń w SDL2.
479  */
480  switch( this->Event.type ){
481  /*
482  Zamknięcie okna krzyżykiem w narożniku.
483  */
484  case SDL_QUIT:
485  this->CheckInit = false;
486  break;
487  /*
488  Akcja poruszenia myszki.
489  */
490  case SDL_MOUSEMOTION:
491  /*
492  Różnica pomiędzy ostaną i aktualną pozycją myszki.
493  */
494  this->Mouse.x = -this->Event.motion.xrel;
495  this->Mouse.y = -this->Event.motion.yrel;
496  /*
497  Wyślij to do kamery.
498  */
499  this->camera.MouseUpdate( this->Mouse );
500  /*
501  Ustaw myszkę na środku ekranu.
502  */
503  SDL_WarpMouseInWindow( this->Okno, ( this->Wysokosc / 2 ), ( this->Szerokosc / 2 ) );
504  break;
505  /*
506  Akcja wciśnięcia klawisza.
507  */
508  case SDL_KEYDOWN:
509  switch( this->Event.key.keysym.sym ){
510  /*
511  ESC - wyjście z aplikacji.
512  */
513  case SDLK_ESCAPE:
514  this->CheckInit = false;
515  break;
516  /*
517  Poruszenie kamery do przodu (prze siebie).
518  */
519  case SDLK_UP:
520  case SDLK_w:
521  this->camera.MoveForward();
522  break;
523  /*
524  Poruszenie kamery do tyłu.
525  */
526  case SDLK_DOWN:
527  case SDLK_s:
528  this->camera.MoveBackward();
529  break;
530  /*
531  Poruszenie kamery w lewo.
532  */
533  case SDLK_LEFT:
534  case SDLK_a:
535  this->camera.MoveLeft();
536  break;
537  /*
538  Poruszenie kamery w prawo.
539  */
540  case SDLK_RIGHT:
541  case SDLK_d:
542  this->camera.MoveRight();
543  break;
544  /*
545  Poruszenie kamery do góry.

547  case SDLK_SPACE:

549  break;

551  Poruszenie kamery w dół.

564  SDL_SetRelativeMouseMode( SDL_TRUE );

573  }

577  }
578  }

602 
603  W kolejności:
604  - macierz widoku
605  - macierz projekcji
606
