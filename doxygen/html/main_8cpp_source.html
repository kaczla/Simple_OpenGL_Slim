317 void Gra::Start(){
318  while( this->CheckInit ){
319  while( SDL_PollEvent( & this->Event ) ){
320  switch( this->Event.type ){
321  case SDL_QUIT:
322  this->CheckInit = false;
323  break;
324  case SDL_MOUSEMOTION:
325  this->Mouse.x = -this->Event.motion.xrel;
326  this->Mouse.y = -this->Event.motion.yrel;
327  this->camera.MouseUpdate( this->Mouse );
328  SDL_WarpMouseInWindow( this->Okno, ( this->Wysokosc / 2 ), ( this->Szerokosc / 2 ) );
329  break;
330  case SDL_KEYDOWN:
331  switch( this->Event.key.keysym.sym ){
332  case SDLK_ESCAPE:
333  this->CheckInit = false;
334  break;
335  case SDLK_UP:
336  case SDLK_w:
337  this->camera.MoveForward();
338  break;
339  case SDLK_DOWN:
340  case SDLK_s:
341  this->camera.MoveBackward();
342  break;
343  case SDLK_LEFT:
344  case SDLK_a:
345  this->camera.MoveLeft();
346  break;
347  case SDLK_RIGHT:
348  case SDLK_d:
349  this->camera.MoveRight();
350  break;
351  case SDLK_SPACE:
352  this->camera.MoveUp();
353  break;
354  case SDLK_LCTRL:
355  case SDLK_c:
